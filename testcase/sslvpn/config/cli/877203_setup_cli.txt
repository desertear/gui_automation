config user local
    edit "${username}"
        set type password
        set passwd ${password}
    next
end

config user group
    edit "${group_name}"
        set member "${username}"
    next
end

config vpn ssl settings
    config authentication-rule
        edit 1002
            set groups "${group_name}"
            set portal "${FGT_SSLVPN_PORTAL_NAME}"
        next
    end
end

config firewall policy
    edit ${FGT_POLICY_ID_FOR_SSLVPN}
        append groups "${group_name}"
    next
end

config vpn ssl web portal
    edit "${FGT_SSLVPN_PORTAL_NAME}"
        config bookmark-group
            edit "gui-bookmarks"
                config bookmarks
                    edit "${bookmark_name}"
                        set apptype ftp
                        set description "this is tmp ftp bookmark"
                        set folder "${folder}"
                        set sso auto
                        set sso-credential sslvpn-login
                    next
                end
            next
        end
    next
end
